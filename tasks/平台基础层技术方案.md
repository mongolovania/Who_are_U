```dart
// 采用 BLoC 模式 + Repository 模式
//
// 数据流:
//   UI (Widget) → Cubit/BLoC (emit State) → UI rebuild
//                     │
//                     ▼
//               Repository (业务逻辑)
//                     │
//               ┌─────┴─────┐
//               ▼           ▼
//          Local DS    Remote DS
//        (SQLite·File) (HTTP API·Sync)

// 全局依赖注入 (GetIt)
final sl = GetIt.instance;

Future<void> initDependencies() async {
  // Core
  sl.registerLazySingleton<Database>(() => Database.init());
  sl.registerLazySingleton<SecureStore>(() => SecureStore());
  sl.registerLazySingleton<FileStore>(() => FileStore(sl()));
  sl.registerLazySingleton<ApiClient>(() => ApiClient());

  // Repositories
  sl.registerLazySingleton<PersonaRepository>(() => PersonaRepository(sl(), sl()));
  sl.registerLazySingleton<ConversationRepository>(() => ConversationRepository(sl(), sl(), sl()));
  sl.registerLazySingleton<MountainRepository>(() => MountainRepository(sl()));
  sl.registerLazySingleton<WardrobeRepository>(() => WardrobeRepository(sl()));
  sl.registerLazySingleton<SubscriptionRepository>(() => SubscriptionRepository(sl()));

  // Registries (热加载·静态数据)
  sl.registerLazySingleton<ArchetypeRegistry>(() => ArchetypeRegistry());
  sl.registerLazySingleton<MasterRegistry>(() => MasterRegistry()..load());
}
```

```python
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

from app.db.redis import init_redis, close_redis

from app.middleware.auth_middleware import AuthMiddleware

    # Startup
    await init_db(settings.DATABASE_URL)
    await init_redis(settings.REDIS_URL)
    # 加载大师数据
    from app.services.master_registry import master_registry

    # Shutdown
    await close_db()
    await close_redis()

    version="0.1.0",
```

```python
# ai_service.py — LLM 统一调用封装
class AIService:
    def __init__(self):
        self.deepseek_client = AsyncOpenAI(
            base_url="https://api.deepseek.com/v1",
            api_key=settings.DEEPSEEK_API_KEY,
            timeout=60.0,
        )
        self.gpt4o_client = AsyncOpenAI(
            api_key=settings.OPENAI_API_KEY,
            timeout=60.0,
        )
        self.token_counter = TokenCounter()

    async def chat(
        self,
        messages: list[dict],
        mode: ChatMode,                # "chat" | "decision"
        archetypes: list[str],
        active_masters: list[str],
        breath_context: str | None = None,
        model: str = "deepseek-chat",  # 默认 DeepSeek-V3
        stream: bool = True,
    ) -> AsyncIterator[ChatCompletionChunk]:
        """对话接口"""
        system_prompt = PromptTemplates.build(
            mode=mode,
            archetypes=archetypes,
            active_masters=active_masters,
            breath_context=breath_context,
        )
        client = self._get_client(model)
        response = await client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": system_prompt},
                *messages,
            ],
            stream=stream,
            max_tokens=2048,
            temperature=0.7 if mode == ChatMode.CHAT else 0.5,
        )
        # 流式输出 + token 计数
        ...

    def _get_client(self, model: str):
        if model.startswith("gpt"):
            return self.gpt4o_client
        return self.deepseek_client
```

### 4.1 SQLite + sqlcipher

```dart
class Database {
  static Database? _instance;
  late final sql.Database _db;

  static Future<Database> init() async {
    if (_instance != null) return _instance!;

    // 获取或生成加密密钥
    final key = await SecureStore.getOrCreate('db_encryption_key');

    _instance = Database._();
    _instance!._db = await sql.openDatabase(
      'whoareu.db',
      password: key,  // sqlcipher 加密
      version: 1,
      onCreate: _onCreate,
      onUpgrade: _onUpgrade,
    );
    return _instance!;
  }

  static Future<void> _onCreate(sql.Database db, int version) async {
```

### 4.2 加密文件存储

```dart
class FileStore {
  final Database _db;
  final String _basePath;

  /// 写入加密文件
  Future<File> write(String filename, String content) async {
    final key = await _getEncryptionKey();
    final nonce = CryptoUtils.generateNonce();  // 12 bytes

    final ciphertext = AesGcm.encrypt(
      plaintext: utf8.encode(content),
      key: key,
      nonce: nonce,
    );

    // 文件格式: [nonce (12 bytes)] [ciphertext + tag]
    final output = Uint8List.fromList([...nonce, ...ciphertext]);
    final file = File('$_basePath/$filename.enc');
    await file.writeAsBytes(output);
    return file;
  }

  /// 读取加密文件
  Future<String> read(String filename) async {
    final key = await _getEncryptionKey();
    final file = File('$_basePath/$filename.enc');
    final data = await file.readAsBytes();

    // 解析: 前12字节 = nonce
    final nonce = data.sublist(0, 12);
    final ciphertext = data.sublist(12);

    final plaintext = AesGcm.decrypt(
      ciphertext: ciphertext,
      key: key,
      nonce: nonce,
    );
    return utf8.decode(plaintext);
  }

  Future<Uint8List> _getEncryptionKey() async {
    // 从 SecureStore 获取文件加密密钥
    return await SecureStore.getOrCreate('file_encryption_key');
  }

  /// 文件命名约定
  String conversationPath(String id) => 'conv_$id.enc';
  String reportPath(String id) => 'report_$id.enc';
  String ombreBucketPath(String bucketId) => 'ombre_$bucketId.enc';
}
```

```python
class ChatSSEClient {
  StreamSubscription? _subscription;
  int _retryCount = 0;
  static const _maxRetries = 3;
  static const _retryDelay = Duration(seconds: 2);

  Future<void> connect(String conversationId) async {
    try {
      final response = await dio.post(
        '/api/chat',
        data: request.toJson(),
        options: Options(responseType: ResponseType.stream),
      );

      _subscription = response.data.stream
        .transform(utf8.decoder)
        .transform(LineSplitter())
        .listen(_onData, onError: _onError, onDone: _onDone);
```

### 5.5 超时与重试

```python
class AIService:
    RETRY_CONFIG = {
        "max_retries": 3,
        "base_delay": 1.0,       # 初始延迟 1s
        "max_delay": 10.0,       # 最大延迟 10s
        "backoff_factor": 2.0,   # 指数退避
        "retryable_errors": [
            "rate_limit",
            "server_error",
            "timeout",
        ],
    }

    async def _call_with_retry(self, client, **kwargs):
        for attempt in range(self.RETRY_CONFIG["max_retries"]):
            try:
                return await client.chat.completions.create(
                    timeout=30.0,
                    **kwargs,
                )
            except Exception as e:
                if attempt == self.RETRY_CONFIG["max_retries"] - 1:
                    raise
                delay = min(
                    self.RETRY_CONFIG["base_delay"] *
                    (self.RETRY_CONFIG["backoff_factor"] ** attempt),
                    self.RETRY_CONFIG["max_delay"],
                )
                await asyncio.sleep(delay)
```

### 5.7 Token 计数与成本追踪

```python
class TokenCounter:
    def __init__(self):
        self._daily_tokens = defaultdict(int)    # user_id → daily tokens
        self._monthly_cost = defaultdict(float)  # user_id → monthly cost

    async def count_and_track(
        self,
        user_id: str,
        model: str,
        input_tokens: int,
        output_tokens: int,
    ):
        """统计 token 用量并追踪成本"""
        cost = self._calculate_cost(model, input_tokens, output_tokens)

        # Redis 原子增量
        await redis.incrby(f"tokens:daily:{user_id}", input_tokens + output_tokens)
        await redis.expire(f"tokens:daily:{user_id}", 86400)
        await redis.incrbyfloat(f"cost:monthly:{user_id}", cost)
        await redis.expire(f"cost:monthly:{user_id}", 2592000)  # 30 days

        # 成本阈值告警（超重度用户）
        monthly = await redis.get(f"cost:monthly:{user_id}")
        if float(monthly or 0) > settings.MAX_MONTHLY_COST_PER_USER:
            logger.warning(f"用户 {user_id} 月度成本超限: ¥{monthly}")

    def _calculate_cost(self, model: str, input_tokens: int, output_tokens: int) -> float:
        if model == "deepseek-chat":
            return (input_tokens * 1.0 + output_tokens * 2.0) / 1_000_000
        elif model == "gpt-4o":
            return (input_tokens * 17.5 + output_tokens * 52.5) / 1_000_000
        return 0.0
```
